 39class _Dialect(type):
 40    classes: t.Dict[str, Dialect] = {}

 42    @classmethod
 43    def __getitem__(cls, key):
 44        return cls.classes[key]

 46    @classmethod
 47    def get(cls, key, default=None):
 48        return cls.classes.get(key, default)

 50    def __new__(cls, clsname, bases, attrs):
 51        klass = super().__new__(cls, clsname, bases, attrs)
 52        enum = Dialects.__members__.get(clsname.upper())
 53        cls.classes[enum.value if enum is not None else clsname.lower()] = klass
 54
 55        klass.time_trie = new_trie(klass.time_mapping)
 56        klass.inverse_time_mapping = {v: k for k, v in klass.time_mapping.items()}
 57        klass.inverse_time_trie = new_trie(klass.inverse_time_mapping)
 58
 59        klass.tokenizer_class = getattr(klass, "Tokenizer", Tokenizer)
 60        klass.parser_class = getattr(klass, "Parser", Parser)
 61        klass.generator_class = getattr(klass, "Generator", Generator)
 62
 63        klass.quote_start, klass.quote_end = list(klass.tokenizer_class._QUOTES.items())[0]
 64        klass.identifier_start, klass.identifier_end = list(
 65            klass.tokenizer_class._IDENTIFIERS.items()
 66        )[0]
 67
 68        if (
 69            klass.tokenizer_class._BIT_STRINGS
 70            and exp.BitString not in klass.generator_class.TRANSFORMS
 71        ):
 72            bs_start, bs_end = list(klass.tokenizer_class._BIT_STRINGS.items())[0]
 73            klass.generator_class.TRANSFORMS[
 74                exp.BitString
 75            ] = lambda self, e: f"{bs_start}{int(self.sql(e, 'this')):b}{bs_end}"
 76        if (
 77            klass.tokenizer_class._HEX_STRINGS
 78            and exp.HexString not in klass.generator_class.TRANSFORMS
 79        ):
 80            hs_start, hs_end = list(klass.tokenizer_class._HEX_STRINGS.items())[0]
 81            klass.generator_class.TRANSFORMS[
 82                exp.HexString
 83            ] = lambda self, e: f"{hs_start}{int(self.sql(e, 'this')):X}{hs_end}"
 84        if (
 85            klass.tokenizer_class._BYTE_STRINGS
 86            and exp.ByteString not in klass.generator_class.TRANSFORMS

 88            be_start, be_end = list(klass.tokenizer_class._BYTE_STRINGS.items())[0]
 89            klass.generator_class.TRANSFORMS[
 90                exp.ByteString
 91            ] = lambda self, e: f"{be_start}{self.sql(e, 'this')}{be_end}"

 96class Dialect(metaclass=_Dialect):

 98    unnest_column_only = False

121    @classmethod
122    def get_or_raise(cls, dialect):
123        if not dialect:
124            return cls
125        if isinstance(dialect, _Dialect):
126            return dialect
127        if isinstance(dialect, Dialect):
128            return dialect.__class__

130        result = cls.get(dialect)
131        if not result:
132            raise ValueError(f"Unknown dialect '{dialect}'")

136    @classmethod
137    def format_time(cls, expression):
138        if isinstance(expression, str):
139            return exp.Literal.string(
140                format_time(
141                    expression[1:-1],  # the time formats are quoted
142                    cls.time_mapping,
143                    cls.time_trie,
144                )

146        if expression and expression.is_string:
147            return exp.Literal.string(
148                format_time(
149                    expression.this,
150                    cls.time_mapping,
151                    cls.time_trie,

302def var_map_sql(self, expression, map_func_name="MAP"):
303    keys = expression.args["keys"]
304    values = expression.args["values"]
305
306    if not isinstance(keys, exp.Array) or not isinstance(values, exp.Array):
307        self.unsupported("Cannot convert array columns into map.")
308        return f"{map_func_name}({self.format_args(keys, values)})"
309
310    args = []
311    for key, value in zip(keys.expressions, values.expressions):
312        args.append(self.sql(key))
313        args.append(self.sql(value))
314    return f"{map_func_name}({self.format_args(*args)})"

318    """Helper used for time expressions.
319
320    Args
321        exp_class (Class): the expression class to instantiate

337def create_with_partitions_sql(self, expression):
338    """
339    In Hive and Spark, the PARTITIONED BY property acts as an extension of a table's schema. When the
340    PARTITIONED BY value is an array of column names, they are transformed into a schema. The corresponding
341    columns are removed from the create statement.
342    """
343    has_schema = isinstance(expression.this, exp.Schema)
344    is_partitionable = expression.args.get("kind") in ("TABLE", "VIEW")
345
346    if has_schema and is_partitionable:
347        expression = expression.copy()
348        prop = expression.find(exp.PartitionedByProperty)
349        this = prop and prop.this
350        if prop and not isinstance(this, exp.Schema):
351            schema = expression.this
352            columns = {v.name.upper() for v in this.expressions}
353            partitions = [col for col in schema.expressions if col.name.upper() in columns]
354            schema.set("expressions", [e for e in schema.expressions if e not in partitions])
355            prop.replace(exp.PartitionedByProperty(this=exp.Schema(expressions=partitions)))
356            expression.set("this", schema)
357
358    return self.create_sql(expression)

361def parse_date_delta(exp_class, unit_mapping=None):
362    def inner_func(args):
363        unit_based = len(args) == 3
364        this = seq_get(args, 2) if unit_based else seq_get(args, 0)
365        expression = seq_get(args, 1) if unit_based else seq_get(args, 1)
366        unit = seq_get(args, 0) if unit_based else exp.Literal.string("DAY")
367        unit = unit_mapping.get(unit.name.lower(), unit) if unit_mapping else unit
368        return exp_class(this=this, expression=expression, unit=unit)
369
370    return inner_func

396def trim_sql(self, expression):
397    target = self.sql(expression, "this")
398    trim_type = self.sql(expression, "position")
399    remove_chars = self.sql(expression, "expression")
400    collation = self.sql(expression, "collation")
401
402    # Use TRIM/LTRIM/RTRIM syntax if the expression isn't database-specific
403    if not remove_chars and not collation:
404        return self.trim_sql(expression)
405
406    trim_type = f"{trim_type} " if trim_type else ""
407    remove_chars = f"{remove_chars} " if remove_chars else ""
408    from_part = "FROM " if trim_type or remove_chars else ""
409    collation = f" COLLATE {collation}" if collation else ""
410    return f"TRIM({trim_type}{remove_chars}{from_part}{target}{collation})"

338def create_with_partitions_sql(self, expression):
339    """
340    In Hive and Spark, the PARTITIONED BY property acts as an extension of a table's schema. When the
341    PARTITIONED BY value is an array of column names, they are transformed into a schema. The corresponding
342    columns are removed from the create statement.
343    """
344    has_schema = isinstance(expression.this, exp.Schema)
345    is_partitionable = expression.args.get("kind") in ("TABLE", "VIEW")
346
347    if has_schema and is_partitionable:
348        expression = expression.copy()
349        prop = expression.find(exp.PartitionedByProperty)
350        this = prop and prop.this
351        if prop and not isinstance(this, exp.Schema):
352            schema = expression.this
353            columns = {v.name.upper() for v in this.expressions}
354            partitions = [col for col in schema.expressions if col.name.upper() in columns]
355            schema.set("expressions", [e for e in schema.expressions if e not in partitions])
356            prop.replace(exp.PartitionedByProperty(this=exp.Schema(expressions=partitions)))
357            expression.set("this", schema)
358
359    return self.create_sql(expression)
