 53def _date_sql(self: DuckDB.Generator, expression: exp.Date) -> str:
 54    result = f"CAST({self.sql(expression, 'this')} AS DATE)"
 55    zone = self.sql(expression, "zone")
 56
 57    if zone:
 58        date_str = self.func("STRFTIME", result, "'%d/%m/%Y'")
 59        date_str = f"{date_str} || ' ' || {zone}"
 60
 61        # This will create a TIMESTAMP with time zone information
 62        result = self.func("STRPTIME", date_str, "'%d/%m/%Y %Z'")
 63
 64    return result

102def _struct_sql(self: DuckDB.Generator, expression: exp.Struct) -> str:
103    args: t.List[str] = []
104    for expr in expression.expressions:
105        if isinstance(expr, exp.Alias):
106            key = expr.alias
107            value = expr.this
108        else:
109            key = expr.name or expr.this.name
110            if isinstance(expr, exp.Bracket):
111                value = expr.expressions[0]
112            else:
113                value = expr.expression
114
115        args.append(f"{self.sql(exp.Literal.string(key))}: {self.sql(value)}")
116
117    return f"{{{', '.join(args)}}}"

136def _unix_to_time_sql(self: DuckDB.Generator, expression: exp.UnixToTime) -> str:
137    scale = expression.args.get("scale")
138    timestamp = self.sql(expression, "this")
139    if scale in (None, exp.UnixToTime.SECONDS):
140        return f"TO_TIMESTAMP({timestamp})"
141    if scale == exp.UnixToTime.MILLIS:
142        return f"EPOCH_MS({timestamp})"
143    if scale == exp.UnixToTime.MICROS:
144        return f"MAKE_TIMESTAMP({timestamp})"
145    if scale == exp.UnixToTime.NANOS:
146        return f"TO_TIMESTAMP({timestamp} / 1000000000)"
147
148    self.unsupported(f"Unsupported scale for timestamp: {scale}.")
149    return ""

Generator converts a given syntax tree to the corresponding SQL string.

Arguments:
  • pretty: Whether or not to format the produced SQL string. Default: False.
  • identify: Determines when an identifier should be quoted. Possible values are: False (default): Never quote, except in cases where it's mandatory by the dialect. True or 'always': Always quote. 'safe': Only quote identifiers that are case insensitive.
  • normalize: Whether or not to normalize identifiers to lowercase. Default: False.
  • pad: Determines the pad size in a formatted string. Default: 2.
  • indent: Determines the indentation size in a formatted string. Default: 2.
  • normalize_functions: Whether or not to normalize all function names. Possible values are: "upper" or True (default): Convert names to uppercase. "lower": Convert names to lowercase. False: Disables function name normalization.
  • unsupported_level: Determines the generator's behavior when it encounters unsupported expressions. Default ErrorLevel.WARN.
  • max_unsupported: Maximum number of unsupported messages to include in a raised UnsupportedError. This is only relevant if unsupported_level is ErrorLevel.RAISE. Default: 3
  • leading_comma: Determines whether or not the comma is leading or trailing in select expressions. This is only relevant when generating in pretty mode. Default: False
  • max_text_width: The max number of characters in a segment before creating new lines in pretty mode. The default is on the smaller end because the length only represents a segment and not the true line length. Default: 80
  • comments: Whether or not to preserve comments in the output SQL code. Default: True
