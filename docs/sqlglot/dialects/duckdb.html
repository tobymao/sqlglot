 52def _date_sql(self: DuckDB.Generator, expression: exp.Date) -> str:
 53    result = f"CAST({self.sql(expression, 'this')} AS DATE)"
 54    zone = self.sql(expression, "zone")
 55
 56    if zone:
 57        date_str = self.func("STRFTIME", result, "'%d/%m/%Y'")
 58        date_str = f"{date_str} || ' ' || {zone}"
 59
 60        # This will create a TIMESTAMP with time zone information
 61        result = self.func("STRPTIME", date_str, "'%d/%m/%Y %Z'")
 62
 63    return result

101def _struct_sql(self: DuckDB.Generator, expression: exp.Struct) -> str:
102    args: t.List[str] = []
103    for expr in expression.expressions:
104        if isinstance(expr, exp.Alias):
105            key = expr.alias
106            value = expr.this
107        else:
108            key = expr.name or expr.this.name
109            if isinstance(expr, exp.Bracket):
110                value = expr.expressions[0]
111            else:
112                value = expr.expression
113
114        args.append(f"{self.sql(exp.Literal.string(key))}: {self.sql(value)}")
115
116    return f"{{{', '.join(args)}}}"

135def _unix_to_time_sql(self: DuckDB.Generator, expression: exp.UnixToTime) -> str:
136    scale = expression.args.get("scale")
137    timestamp = self.sql(expression, "this")
138    if scale in (None, exp.UnixToTime.SECONDS):
139        return f"TO_TIMESTAMP({timestamp})"
140    if scale == exp.UnixToTime.MILLIS:
141        return f"EPOCH_MS({timestamp})"
142    if scale == exp.UnixToTime.MICROS:
143        return f"MAKE_TIMESTAMP({timestamp})"
144    if scale == exp.UnixToTime.NANOS:
145        return f"TO_TIMESTAMP({timestamp} / 1000000000)"
146
147    self.unsupported(f"Unsupported scale for timestamp: {scale}.")
148    return ""

412        def interval_sql(self, expression: exp.Interval) -> str:
413            multiplier: t.Optional[int] = None
414            unit = expression.text("unit").lower()
415
416            if unit.startswith("week"):
417                multiplier = 7
418            if unit.startswith("quarter"):
419                multiplier = 90
420
421            if multiplier:
422                return f"({multiplier} * {super().interval_sql(exp.Interval(this=expression.this, unit=exp.var('day')))})"
423
424            return super().interval_sql(expression)

Generator converts a given syntax tree to the corresponding SQL string.

Arguments:
  • pretty: Whether or not to format the produced SQL string. Default: False.
  • identify: Determines when an identifier should be quoted. Possible values are: False (default): Never quote, except in cases where it's mandatory by the dialect. True or 'always': Always quote. 'safe': Only quote identifiers that are case insensitive.
  • normalize: Whether or not to normalize identifiers to lowercase. Default: False.
  • pad: Determines the pad size in a formatted string. Default: 2.
  • indent: Determines the indentation size in a formatted string. Default: 2.
  • normalize_functions: Whether or not to normalize all function names. Possible values are: "upper" or True (default): Convert names to uppercase. "lower": Convert names to lowercase. False: Disables function name normalization.
  • unsupported_level: Determines the generator's behavior when it encounters unsupported expressions. Default ErrorLevel.WARN.
  • max_unsupported: Maximum number of unsupported messages to include in a raised UnsupportedError. This is only relevant if unsupported_level is ErrorLevel.RAISE. Default: 3
  • leading_comma: Determines whether or not the comma is leading or trailing in select expressions. This is only relevant when generating in pretty mode. Default: False
  • max_text_width: The max number of characters in a segment before creating new lines in pretty mode. The default is on the smaller end because the length only represents a segment and not the true line length. Default: 80
  • comments: Whether or not to preserve comments in the output SQL code. Default: True
