 78def _add_local_prefix_for_aliases(expression: exp.Expression) -> exp.Expression:
 79    if isinstance(expression, exp.Select):
 80        aliases: dict[str, bool] = {
 81            alias.name: bool(alias.args.get("quoted"))
 82            for sel in expression.selects
 83            if isinstance(sel, exp.Alias) and (alias := sel.args.get("alias"))
 84        }
 85
 86        table = expression.find(exp.Table)
 87        table_ident = table.this if table else None
 88
 89        if (
 90            table_ident
 91            and table_ident.name.upper() == "LOCAL"
 92            and not bool(table_ident.args.get("quoted"))
 93        ):
 94            table_ident.replace(exp.to_identifier(table_ident.name.upper(), quoted=True))
 95
 96        def prefix_local(node, visible_aliases: dict[str, bool]) -> exp.Expression:
 97            if isinstance(node, exp.Column) and not node.table:
 98                if node.name in visible_aliases:
 99                    return exp.Column(
100                        this=exp.to_identifier(node.name, quoted=visible_aliases[node.name]),
101                        table=exp.to_identifier("LOCAL", quoted=False),
102                    )
103            return node
104
105        for key in ("where", "group", "having"):
106            if arg := expression.args.get(key):
107                expression.set(key, arg.transform(lambda node: prefix_local(node, aliases)))
108
109        seen_aliases: dict[str, bool] = {}
110        new_selects: list[exp.Expression] = []
111        for sel in expression.selects:
112            if isinstance(sel, exp.Alias):
113                inner = sel.this.transform(lambda node: prefix_local(node, seen_aliases))
114                sel.set("this", inner)
115
116                alias_node = sel.args.get("alias")
117
118                seen_aliases[sel.alias] = bool(alias_node and getattr(alias_node, "quoted", False))
119                new_selects.append(sel)
120            else:
121                new_selects.append(sel.transform(lambda node: prefix_local(node, seen_aliases)))
122        expression.set("expressions", new_selects)
123
124    return expression

152def _substring_index_sql(self: Exasol.Generator, expression: exp.SubstringIndex) -> str:
153    this = expression.this
154    delimiter = expression.args["delimiter"]
155    count_node = expression.args["count"]
156    count_sql = self.sql(expression, "count")
157    num = count_node.to_py() if count_node.is_number else 0
158
159    haystack_sql = self.sql(this)
160    if num == 0:
161        return self.func("SUBSTR", haystack_sql, "1", "0")
162
163    from_right = num < 0
164    direction = "-1" if from_right else "1"
165    occur = self.func("ABS", count_sql) if from_right else count_sql
166
167    delimiter_sql = self.sql(delimiter)
168
169    position = self.func(
170        "INSTR",
171        self.func("LOWER", haystack_sql) if is_case_insensitive(this) else haystack_sql,
172        self.func("LOWER", delimiter_sql) if is_case_insensitive(delimiter) else delimiter_sql,
173        direction,
174        occur,
175    )
176    nullable_pos = self.func("NULLIF", position, "0")
177
178    if from_right:
179        start = self.func(
180            "NVL", f"{nullable_pos} + {self.func('LENGTH', delimiter_sql)}", direction
181        )
182        return self.func("SUBSTR", haystack_sql, start)
183
184    length = self.func("NVL", f"{nullable_pos} - 1", self.func("LENGTH", haystack_sql))
185    return self.func("SUBSTR", haystack_sql, direction, length)

189def _qualify_unscoped_star(expression: exp.Expression) -> exp.Expression:
190    """
191    Exasol doesn't support a bare * alongside other select items, so we rewrite it
192    Rewrite: SELECT *, <other> FROM <Table>
193    Into: SELECT T.*, <other> FROM <Table> AS T
194    """
195
196    if not isinstance(expression, exp.Select):
197        return expression
198
199    select_expressions = expression.expressions or []
200
201    def is_bare_star(expr: exp.Expression) -> bool:
202        return isinstance(expr, exp.Star) and expr.this is None
203
204    has_other_expression = False
205    bare_star_expr: exp.Expression | None = None
206    for expr in select_expressions:
207        has_bare_star = is_bare_star(expr)
208        if has_bare_star and bare_star_expr is None:
209            bare_star_expr = expr
210        elif not has_bare_star:
211            has_other_expression = True
212        if bare_star_expr and has_other_expression:
213            break
214
215    if not (bare_star_expr and has_other_expression):
216        return expression
217
218    scope = build_scope(expression)
219
220    if not scope or not scope.selected_sources:
221        return expression
222
223    table_identifiers: list[exp.Identifier] = []
224
225    for source_name, (source_expr, _) in scope.selected_sources.items():
226        ident = (
227            source_expr.this.copy()
228            if isinstance(source_expr, exp.Table) and isinstance(source_expr.this, exp.Identifier)
229            else exp.to_identifier(source_name)
230        )
231        table_identifiers.append(ident)
232
233    qualified_star_columns = [
234        exp.Column(this=bare_star_expr.copy(), table=ident) for ident in table_identifiers
235    ]
236
237    new_select_expressions: list[exp.Expression] = []
238
239    for select_expr in select_expressions:
240        new_select_expressions.extend(qualified_star_columns) if is_bare_star(
241            select_expr
242        ) else new_select_expressions.append(select_expr)
243
244    expression.set("expressions", new_select_expressions)
245    return expression

248def _add_date_sql(self: Exasol.Generator, expression: DATE_ADD_OR_SUB) -> str:
249    interval = expression.expression if isinstance(expression.expression, exp.Interval) else None
250
251    unit = (
252        (interval.text("unit") or "DAY").upper()
253        if interval is not None
254        else (expression.text("unit") or "DAY").upper()
255    )
256
257    if unit not in DATE_UNITS:
258        self.unsupported(f"'{unit}' is not supported in Exasol.")
259        return self.function_fallback_sql(expression)
260
261    offset_expr: exp.Expression = expression.expression
262    if interval is not None:
263        offset_expr = interval.this
264
265    if isinstance(expression, exp.DateSub):
266        offset_expr = exp.Neg(this=offset_expr)
267
268    return self.func(f"ADD_{unit}S", expression.this, offset_expr)

Arguments:
  • pretty: Whether to format the produced SQL string. Default: False.
  • identify: Determines when an identifier should be quoted. Possible values are: False (default): Never quote, except in cases where it's mandatory by the dialect. True: Always quote except for specials cases. 'safe': Only quote identifiers that are case insensitive.
  • normalize: Whether to normalize identifiers to lowercase. Default: False.
  • pad: The pad size in a formatted string. For example, this affects the indentation of a projection in a query, relative to its nesting level. Default: 2.
  • indent: The indentation size in a formatted string. For example, this affects the indentation of subqueries and filters under a WHERE clause. Default: 2.
  • normalize_functions: How to normalize function names. Possible values are: "upper" or True (default): Convert names to uppercase. "lower": Convert names to lowercase. False: Disables function name normalization.
  • unsupported_level: Determines the generator's behavior when it encounters unsupported expressions. Default ErrorLevel.WARN.
  • max_unsupported: Maximum number of unsupported messages to include in a raised UnsupportedError. This is only relevant if unsupported_level is ErrorLevel.RAISE. Default: 3
  • leading_comma: Whether the comma is leading or trailing in select expressions. This is only relevant when generating in pretty mode. Default: False
  • max_text_width: The max number of characters in a segment before creating new lines in pretty mode. The default is on the smaller end because the length only represents a segment and not the true line length. Default: 80
  • comments: Whether to preserve comments in the output SQL code. Default: True
