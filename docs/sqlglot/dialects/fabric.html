 11class Fabric(TSQL):
 12    """
 13    Microsoft Fabric Data Warehouse dialect that inherits from T-SQL.
 14
 15    Microsoft Fabric is a cloud-based analytics platform that provides a unified
 16    data warehouse experience. While it shares much of T-SQL's syntax, it has
 17    specific differences and limitations that this dialect addresses.
 18
 19    Key differences from T-SQL:
 20    - Case-sensitive identifiers (unlike T-SQL which is case-insensitive)
 21    - Limited data type support with mappings to supported alternatives
 22    - Temporal types (DATETIME2, DATETIMEOFFSET, TIME) limited to 6 digits precision
 23    - Certain legacy types (MONEY, SMALLMONEY, etc.) are not supported
 24    - Unicode types (NCHAR, NVARCHAR) are mapped to non-unicode equivalents
 25
 26    References:
 27    - Data Types: https://learn.microsoft.com/en-us/fabric/data-warehouse/data-types
 28    - T-SQL Surface Area: https://learn.microsoft.com/en-us/fabric/data-warehouse/tsql-surface-area
 29    """

 31    # Fabric is case-sensitive unlike T-SQL which is case-insensitive
 32    NORMALIZATION_STRATEGY = NormalizationStrategy.CASE_SENSITIVE
 33
 34    class Tokenizer(TSQL.Tokenizer):
 35        # Override T-SQL tokenizer to handle TIMESTAMP differently
 36        # In T-SQL, TIMESTAMP is a synonym for ROWVERSION, but in Fabric we want it to be a datetime type
 37        # Also add UTINYINT keyword mapping since T-SQL doesn't have it
 38        KEYWORDS = {
 39            **TSQL.Tokenizer.KEYWORDS,
 40            "TIMESTAMP": TokenType.TIMESTAMP,
 41            "UTINYINT": TokenType.UTINYINT,
 42        }

 47        TYPE_MAPPING = {
 48            **TSQL.Generator.TYPE_MAPPING,
 49            exp.DataType.Type.DATETIME: "DATETIME2",
 50            exp.DataType.Type.DECIMAL: "DECIMAL",
 51            exp.DataType.Type.IMAGE: "VARBINARY",

 54            exp.DataType.Type.MONEY: "DECIMAL",

 57            exp.DataType.Type.ROWVERSION: "ROWVERSION",
 58            exp.DataType.Type.SMALLDATETIME: "DATETIME2",
 59            exp.DataType.Type.SMALLMONEY: "DECIMAL",
 60            exp.DataType.Type.TIMESTAMP: "DATETIME2",
 61            exp.DataType.Type.TIMESTAMPNTZ: "DATETIME2",
 62            exp.DataType.Type.TIMESTAMPTZ: "DATETIMEOFFSET",
 63            exp.DataType.Type.TINYINT: "SMALLINT",

 65            exp.DataType.Type.UUID: "VARBINARY(MAX)",
 66            exp.DataType.Type.XML: "VARCHAR",
 67        }

 69        def datatype_sql(self, expression: exp.DataType) -> str:
 70            # Check if this is a temporal type that needs precision handling. Fabric limits temporal
 71            # types to max 6 digits precision. When no precision is specified, we default to 6 digits.
 72            if (
 73                expression.is_type(*exp.DataType.TEMPORAL_TYPES)
 74                and expression.this != exp.DataType.Type.DATE
 75            ):
 76                # Get the current precision (first expression if it exists)
 77                precision_param = expression.find(exp.DataTypeParam)
 78                target_precision = 6
 79
 80                if precision_param and precision_param.this.is_int:
 81                    # Cap precision at 6
 82                    current_precision = precision_param.this.to_py()
 83                    target_precision = min(current_precision, 6)
 84                else:
 85                    # If precision exists but is not an integer, default to 6
 86                    target_precision = 6
 87
 88                # Create a new expression with the target precision
 89                expression = exp.DataType(
 90                    this=expression.this,
 91                    expressions=[exp.DataTypeParam(this=exp.Literal.number(target_precision))],
 92                )
 93
 94            return super().datatype_sql(expression)
 95
 96        def cast_sql(self, expression: exp.Cast, safe_prefix: t.Optional[str] = None) -> str:
 97            # Special handling for TIMESTAMPTZ casts in Fabric
 98            if expression.to and expression.to.is_type(exp.DataType.Type.TIMESTAMPTZ):
 99                attimezone = expression.find_ancestor(exp.AtTimeZone, exp.Select)
100                if not isinstance(attimezone, exp.AtTimeZone):
101                    # We're not inside an AT TIME ZONE, so wrap the cast in AT TIME ZONE 'UTC'
102                    at_time_zone = exp.AtTimeZone(this=expression, zone=exp.Literal.string("UTC"))
103                    return self.sql(at_time_zone)
104
105            return super().cast_sql(expression, safe_prefix)
106
107        def unixtotime_sql(self, expression: exp.UnixToTime) -> str:
108            scale = expression.args.get("scale")
109            timestamp = expression.this
110
111            if scale not in (None, exp.UnixToTime.SECONDS):
112                self.unsupported(f"UnixToTime scale {scale} is not supported by Fabric")
113                return ""
114
115            # Convert unix timestamp (seconds) to microseconds and round to avoid decimals
116            microseconds = timestamp * exp.Literal.number("1e6")
117            rounded = exp.func("round", microseconds, 0)
118            rounded_ms_as_bigint = exp.cast(rounded, exp.DataType.Type.BIGINT)
119
120            # Create the base datetime as '1970-01-01' cast to DATETIME2(6)
121            epoch_start = exp.cast("'1970-01-01'", "datetime2(6)", dialect="fabric")
122
123            dateadd = exp.DateAdd(
124                this=epoch_start,
125                expression=rounded_ms_as_bigint,
126                unit=exp.Literal.string("MICROSECONDS"),
127            )
128            return self.sql(dateadd)

Microsoft Fabric Data Warehouse dialect that inherits from T-SQL.

Microsoft Fabric is a cloud-based analytics platform that provides a unified data warehouse experience. While it shares much of T-SQL's syntax, it has specific differences and limitations that this dialect addresses.

Key differences from T-SQL:

  • Case-sensitive identifiers (unlike T-SQL which is case-insensitive)
  • Limited data type support with mappings to supported alternatives
  • Temporal types (DATETIME2, DATETIMEOFFSET, TIME) limited to 6 digits precision
  • Certain legacy types (MONEY, SMALLMONEY, etc.) are not supported
  • Unicode types (NCHAR, NVARCHAR) are mapped to non-unicode equivalents

References:

  • Data Types: https://learn.microsoft.com/en-us/fabric/data-warehouse/data-types
  • T-SQL Surface Area: https://learn.microsoft.com/en-us/fabric/data-warehouse/tsql-surface-area

Generator converts a given syntax tree to the corresponding SQL string.

Arguments:
  • pretty: Whether to format the produced SQL string. Default: False.
  • identify: Determines when an identifier should be quoted. Possible values are: False (default): Never quote, except in cases where it's mandatory by the dialect. True or 'always': Always quote. 'safe': Only quote identifiers that are case insensitive.
  • normalize: Whether to normalize identifiers to lowercase. Default: False.
  • pad: The pad size in a formatted string. For example, this affects the indentation of a projection in a query, relative to its nesting level. Default: 2.
  • indent: The indentation size in a formatted string. For example, this affects the indentation of subqueries and filters under a WHERE clause. Default: 2.
  • normalize_functions: How to normalize function names. Possible values are: "upper" or True (default): Convert names to uppercase. "lower": Convert names to lowercase. False: Disables function name normalization.
  • unsupported_level: Determines the generator's behavior when it encounters unsupported expressions. Default ErrorLevel.WARN.
  • max_unsupported: Maximum number of unsupported messages to include in a raised UnsupportedError. This is only relevant if unsupported_level is ErrorLevel.RAISE. Default: 3
  • leading_comma: Whether the comma is leading or trailing in select expressions. This is only relevant when generating in pretty mode. Default: False
  • max_text_width: The max number of characters in a segment before creating new lines in pretty mode. The default is on the smaller end because the length only represents a segment and not the true line length. Default: 80
  • comments: Whether to preserve comments in the output SQL code. Default: True
