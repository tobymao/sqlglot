 45def _date_add_sql(kind: str) -> t.Callable[[Postgres.Generator, DATE_ADD_OR_SUB], str]:
 46    def func(self: Postgres.Generator, expression: DATE_ADD_OR_SUB) -> str:
 47        if isinstance(expression, exp.TsOrDsAdd):
 48            expression = ts_or_ds_add_cast(expression)
 49
 50        this = self.sql(expression, "this")
 51        unit = expression.args.get("unit")
 52
 53        expression = self._simplify_unless_literal(expression.expression)
 54        if not isinstance(expression, exp.Literal):
 55            self.unsupported("Cannot add non literal")
 56
 57        expression.args["is_string"] = True
 58        return f"{this} {kind} {self.sql(exp.Interval(this=expression, unit=unit))}"

 63def _date_diff_sql(self: Postgres.Generator, expression: exp.DateDiff) -> str:
 64    unit = expression.text("unit").upper()
 65    factor = DATE_DIFF_FACTOR.get(unit)
 66
 67    end = f"CAST({self.sql(expression, 'this')} AS TIMESTAMP)"
 68    start = f"CAST({self.sql(expression, 'expression')} AS TIMESTAMP)"
 69
 70    if factor is not None:
 71        return f"CAST(EXTRACT(epoch FROM {end} - {start}){factor} AS BIGINT)"
 72
 73    age = f"AGE({end}, {start})"
 74
 75    if unit == "WEEK":
 76        unit = f"EXTRACT(days FROM ({end} - {start})) / 7"
 77    elif unit == "MONTH":
 78        unit = f"EXTRACT(year FROM {age}) * 12 + EXTRACT(month FROM {age})"
 79    elif unit == "QUARTER":
 80        unit = f"EXTRACT(year FROM {age}) * 4 + EXTRACT(month FROM {age}) / 3"
 81    elif unit == "YEAR":
 82        unit = f"EXTRACT(year FROM {age})"
 83    else:
 84        unit = age
 85
 86    return f"CAST({unit} AS BIGINT)"

119def _auto_increment_to_serial(expression: exp.Expression) -> exp.Expression:
120    auto = expression.find(exp.AutoIncrementColumnConstraint)
121
122    if auto:
123        expression.args["constraints"].remove(auto.parent)
124        kind = expression.args["kind"]
125
126        if kind.this == exp.DataType.Type.INT:
127            kind.replace(exp.DataType(this=exp.DataType.Type.SERIAL))
128        elif kind.this == exp.DataType.Type.SMALLINT:
129            kind.replace(exp.DataType(this=exp.DataType.Type.SMALLSERIAL))
130        elif kind.this == exp.DataType.Type.BIGINT:
131            kind.replace(exp.DataType(this=exp.DataType.Type.BIGSERIAL))

136def _serial_to_generated(expression: exp.Expression) -> exp.Expression:
137    kind = expression.args.get("kind")
138    if not kind:
139        return expression
140
141    if kind.this == exp.DataType.Type.SERIAL:
142        data_type = exp.DataType(this=exp.DataType.Type.INT)
143    elif kind.this == exp.DataType.Type.SMALLSERIAL:
144        data_type = exp.DataType(this=exp.DataType.Type.SMALLINT)
145    elif kind.this == exp.DataType.Type.BIGSERIAL:
146        data_type = exp.DataType(this=exp.DataType.Type.BIGINT)
147    else:
148        data_type = None
149
150    if data_type:
151        expression.args["kind"].replace(data_type)
152        constraints = expression.args["constraints"]
153        generated = exp.ColumnConstraint(kind=exp.GeneratedAsIdentityColumnConstraint(this=False))
154        notnull = exp.ColumnConstraint(kind=exp.NotNullColumnConstraint())
155
156        if notnull not in constraints:
157            constraints.insert(0, notnull)
158        if generated not in constraints:
159            constraints.insert(0, generated)
160
161    return expression

164def _generate_series(args: t.List) -> exp.Expression:
165    # The goal is to convert step values like '1 day' or INTERVAL '1 day' into INTERVAL '1' day
166    step = seq_get(args, 2)
167
168    if step is None:
169        # Postgres allows calls with just two arguments -- the "step" argument defaults to 1
170        return exp.GenerateSeries.from_arg_list(args)
171
172    if step.is_string:
173        args[2] = exp.to_interval(step.this)
174    elif isinstance(step, exp.Interval) and not step.args.get("unit"):
175        args[2] = exp.to_interval(step.this.this)
176
177    return exp.GenerateSeries.from_arg_list(args)

190def _merge_sql(self: Postgres.Generator, expression: exp.Merge) -> str:
191    def _remove_target_from_merge(expression: exp.Expression) -> exp.Expression:
192        """Remove table refs from columns in when statements."""
193        if isinstance(expression, exp.Merge):
194            alias = expression.this.args.get("alias")
195
196            normalize = (
197                lambda identifier: self.dialect.normalize_identifier(identifier).name
198                if identifier
199                else None
200            )
201
202            targets = {normalize(expression.this.this)}
203
204            if alias:
205                targets.add(normalize(alias.this))
206
207            for when in expression.expressions:
208                when.transform(
209                    lambda node: exp.column(node.this)
210                    if isinstance(node, exp.Column) and normalize(node.args.get("table")) in targets
211                    else node,
212                    copy=False,
213                )
214
215        return expression
216
217    return transforms.preprocess([_remove_target_from_merge])(self, expression)

356        def _parse_operator(self, this: t.Optional[exp.Expression]) -> t.Optional[exp.Expression]:
357            while True:
358                if not self._match(TokenType.L_PAREN):
359                    break
360
361                op = ""
362                while self._curr and not self._match(TokenType.R_PAREN):
363                    op += self._curr.text
364                    self._advance()
365
366                this = self.expression(
367                    exp.Operator,
368                    comments=self._prev_comments,
369                    this=this,
370                    operator=op,
371                    expression=self._parse_bitwise(),
372                )
373
374                if not self._match(TokenType.OPERATOR):
375                    break
376
377            return this
