 40def _unix_to_time_sql(self: Spark2.Generator, expression: exp.UnixToTime) -> str:
 41    scale = expression.args.get("scale")
 42    timestamp = self.sql(expression, "this")
 43    if scale is None:
 44        return f"CAST(FROM_UNIXTIME({timestamp}) AS TIMESTAMP)"
 45    if scale == exp.UnixToTime.SECONDS:
 46        return f"TIMESTAMP_SECONDS({timestamp})"
 47    if scale == exp.UnixToTime.MILLIS:
 48        return f"TIMESTAMP_MILLIS({timestamp})"
 49    if scale == exp.UnixToTime.MICROS:
 50        return f"TIMESTAMP_MICROS({timestamp})"
 51    if scale == exp.UnixToTime.NANOS:
 52        return f"TIMESTAMP_SECONDS({timestamp} / 1000000000)"
 53
 54    self.unsupported(f"Unsupported scale for timestamp: {scale}.")
 55    return ""

 58def _unalias_pivot(expression: exp.Expression) -> exp.Expression:
 59    """
 60    Spark doesn't allow PIVOT aliases, so we need to remove them and possibly wrap a
 61    pivoted source in a subquery with the same alias to preserve the query's semantics.
 62
 63    Example:
 64        >>> from sqlglot import parse_one
 65        >>> expr = parse_one("SELECT piv.x FROM tbl PIVOT (SUM(a) FOR b IN ('x')) piv")
 66        >>> print(_unalias_pivot(expr).sql(dialect="spark"))
 67        SELECT piv.x FROM (SELECT * FROM tbl PIVOT(SUM(a) FOR b IN ('x'))) AS piv
 68    """
 69    if isinstance(expression, exp.From) and expression.this.args.get("pivots"):
 70        pivot = expression.this.args["pivots"][0]
 71        if pivot.alias:
 72            alias = pivot.args["alias"].pop()
 73            return exp.From(
 74                this=expression.this.replace(
 75                    exp.select("*")
 76                    .from_(expression.this.copy(), copy=False)
 77                    .subquery(alias=alias, copy=False)
 78                )
 79            )
 80
 81    return expression

 84def _unqualify_pivot_columns(expression: exp.Expression) -> exp.Expression:
 85    """
 86    Spark doesn't allow the column referenced in the PIVOT's field to be qualified,
 87    so we need to unqualify it.
 88
 89    Example:
 90        >>> from sqlglot import parse_one
 91        >>> expr = parse_one("SELECT * FROM tbl PIVOT (SUM(tbl.sales) FOR tbl.quarter IN ('Q1', 'Q2'))")
 92        >>> print(_unqualify_pivot_columns(expr).sql(dialect="spark"))
 93        SELECT * FROM tbl PIVOT(SUM(tbl.sales) FOR quarter IN ('Q1', 'Q1'))
 94    """
 95    if isinstance(expression, exp.Pivot):
 96        expression.args["field"].transform(
 97            lambda node: exp.column(node.output_name, quoted=node.this.quoted)
 98            if isinstance(node, exp.Column)
 99            else node,
100            copy=False,
101        )
102
103    return expression

Generator converts a given syntax tree to the corresponding SQL string.

Arguments:
  • pretty: Whether or not to format the produced SQL string. Default: False.
  • identify: Determines when an identifier should be quoted. Possible values are: False (default): Never quote, except in cases where it's mandatory by the dialect. True or 'always': Always quote. 'safe': Only quote identifiers that are case insensitive.
  • normalize: Whether or not to normalize identifiers to lowercase. Default: False.
  • pad: Determines the pad size in a formatted string. Default: 2.
  • indent: Determines the indentation size in a formatted string. Default: 2.
  • normalize_functions: Whether or not to normalize all function names. Possible values are: "upper" or True (default): Convert names to uppercase. "lower": Convert names to lowercase. False: Disables function name normalization.
  • unsupported_level: Determines the generator's behavior when it encounters unsupported expressions. Default ErrorLevel.WARN.
  • max_unsupported: Maximum number of unsupported messages to include in a raised UnsupportedError. This is only relevant if unsupported_level is ErrorLevel.RAISE. Default: 3
  • leading_comma: Determines whether or not the comma is leading or trailing in select expressions. This is only relevant when generating in pretty mode. Default: False
  • max_text_width: The max number of characters in a segment before creating new lines in pretty mode. The default is on the smaller end because the length only represents a segment and not the true line length. Default: 80
  • comments: Whether or not to preserve comments in the output SQL code. Default: True
