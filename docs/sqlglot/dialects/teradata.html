136        def _parse_translate(self, strict: bool) -> exp.Expression:
137            this = self._parse_conjunction()
138
139            if not self._match(TokenType.USING):
140                self.raise_error("Expected USING in TRANSLATE")

143                charset_split = self._prev.text.split("_TO_")

257            if kind == "TABLE" and locations.get(exp.Properties.Location.POST_NAME):
258                this_name = self.sql(expression.this, "this")
259                this_properties = self.properties(
