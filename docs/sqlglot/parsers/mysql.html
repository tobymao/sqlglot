 18# All specifiers for time parts (as opposed to date parts)
 19# https://dev.mysql.com/doc/refman/8.0/en/date-and-time-functions.html#function_date-format
 20TIME_SPECIFIERS = {"f", "H", "h", "I", "i", "k", "l", "p", "r", "S", "s", "T"}

 23def _has_time_specifier(date_format: str) -> bool:
 24    i = 0
 25    length = len(date_format)
 26
 27    while i < length:
 28        if date_format[i] == "%":
 29            i += 1
 30            if i < length and date_format[i] in TIME_SPECIFIERS:
 31                return True
 32        i += 1
 33    return False

 36def _str_to_date(args: list) -> exp.StrToDate | exp.StrToTime:
 37    mysql_date_format = seq_get(args, 1)
 38    date_format = Dialect["mysql"].format_time(mysql_date_format)
 39    this = seq_get(args, 0)
 40
 41    if mysql_date_format and _has_time_specifier(mysql_date_format.name):
 42        return exp.StrToTime(this=this, format=date_format)
 43
 44    return exp.StrToDate(this=this, format=date_format)

343    def _parse_index_constraint(self, kind: str | None = None) -> exp.IndexColumnConstraint:
344        if kind:
345            self._match_texts(("INDEX", "KEY"))
346
347        this = self._parse_id_var(any_token=False)
348        index_type = self._match(TokenType.USING) and self._advance_any() and self._prev.text
349        expressions = self._parse_wrapped_csv(self._parse_ordered)
350
351        options = []
352        while True:
353            if self._match_text_seq("KEY_BLOCK_SIZE"):
354                self._match(TokenType.EQ)
355                opt = exp.IndexConstraintOption(key_block_size=self._parse_number())
356            elif self._match(TokenType.USING):
357                opt = exp.IndexConstraintOption(using=self._advance_any() and self._prev.text)
358            elif self._match_text_seq("WITH", "PARSER"):
359                opt = exp.IndexConstraintOption(parser=self._parse_var(any_token=True))
360            elif self._match(TokenType.COMMENT):
361                opt = exp.IndexConstraintOption(comment=self._parse_string())
362            elif self._match_text_seq("VISIBLE"):
363                opt = exp.IndexConstraintOption(visible=True)
364            elif self._match_text_seq("INVISIBLE"):
365                opt = exp.IndexConstraintOption(visible=False)
366            elif self._match_text_seq("ENGINE_ATTRIBUTE"):
367                self._match(TokenType.EQ)
368                opt = exp.IndexConstraintOption(engine_attr=self._parse_string())
369            elif self._match_text_seq("SECONDARY_ENGINE_ATTRIBUTE"):
370                self._match(TokenType.EQ)
371                opt = exp.IndexConstraintOption(secondary_engine_attr=self._parse_string())
372            else:
373                opt = None
374
375            if not opt:
376                break
377
378            options.append(opt)
379
380        return self.expression(
381            exp.IndexColumnConstraint(
382                this=this,
383                expressions=expressions,
384                kind=kind,
385                index_type=index_type,
386                options=options,
387            )
388        )

390    def _parse_show_mysql(
391        self,
392        this: str,
393        target: bool | str = False,
394        full: bool | None = None,
395        global_: bool | None = None,
396    ) -> exp.Show:
397        json = self._match_text_seq("JSON")
398
399        if target:
400            if isinstance(target, str):
401                self._match_text_seq(*target.split(" "))
402            target_id = self._parse_id_var()
403        else:
404            target_id = None
405
406        log = self._parse_string() if self._match_text_seq("IN") else None
407
408        if this in ("BINLOG EVENTS", "RELAYLOG EVENTS"):
409            position = self._parse_number() if self._match_text_seq("FROM") else None
410            db = None
411        else:
412            position = None
413            db = None
414
415            if self._match(TokenType.FROM):
416                db = self._parse_id_var()
417            elif self._match(TokenType.DOT):
418                db = target_id
419                target_id = self._parse_id_var()
420
421        channel = self._parse_id_var() if self._match_text_seq("FOR", "CHANNEL") else None
422
423        like = self._parse_string() if self._match_text_seq("LIKE") else None
424        where = self._parse_where()
425
426        if this == "PROFILE":
427            types = self._parse_csv(lambda: self._parse_var_from_options(self.PROFILE_TYPES))
428            query = self._parse_number() if self._match_text_seq("FOR", "QUERY") else None
429            offset = self._parse_number() if self._match_text_seq("OFFSET") else None
430            limit = self._parse_number() if self._match_text_seq("LIMIT") else None
431        else:
432            types, query = None, None
433            offset, limit = self._parse_oldstyle_limit()
434
435        mutex = True if self._match_text_seq("MUTEX") else None
436        mutex = False if self._match_text_seq("STATUS") else mutex
437
438        for_table = self._parse_id_var() if self._match_text_seq("FOR", "TABLE") else None
439        for_group = self._parse_string() if self._match_text_seq("FOR", "GROUP") else None
440        for_user = self._parse_string() if self._match_text_seq("FOR", "USER") else None
441        for_role = self._parse_string() if self._match_text_seq("FOR", "ROLE") else None
442        into_outfile = self._parse_string() if self._match_text_seq("INTO", "OUTFILE") else None
443
444        return self.expression(
445            exp.Show(

447                target=target_id,
448                full=full,
449                log=log,
450                position=position,
451                db=db,
452                channel=channel,
453                like=like,

462                for_user=for_user,

467            )

525    def _parse_partition_property(
526        self,
527    ) -> exp.Expr | None | list[exp.Expr]:
528        partition_cls: type[exp.Expr] | None = None
529        value_parser = None
530
531        if self._match_text_seq("RANGE"):
532            partition_cls = exp.PartitionByRangeProperty
533            value_parser = self._parse_partition_range_value
534        elif self._match_text_seq("LIST"):
535            partition_cls = exp.PartitionByListProperty
536            value_parser = self._parse_partition_list_value
537
538        if not partition_cls or not value_parser:
539            return None
540
541        partition_expressions = self._parse_wrapped_csv(self._parse_assignment)
542
543        # For Doris and Starrocks
544        if not self._match_text_seq("(", "PARTITION", advance=False):
545            return partition_expressions
546
547        create_expressions = self._parse_wrapped_csv(value_parser)
548

364                opt = exp.IndexConstraintOption(visible=True)

375
376            if not opt:

381        return self.expression(
382            exp.IndexColumnConstraint(
